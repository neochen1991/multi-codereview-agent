```bash
  "entry": "run.py",
  "execution_mode": "subprocess",

    "type": "object",
    "properties": {

      "nesting_depth": {"type": "integer"},
      "complexity_rating": {"type": "string", "enum": ["low", "moderate", "high", "very_high"]},

```

### 5.2 工具运行脚本

```python
# extensions/tools/custom_complexity_analyzer/run.py
#!/usr/bin/env python3
"""圈复杂度分析工具"""

import json
import re
import sys


def calculate_cyclomatic_complexity(code: str) -> int:
    """计算圈复杂度"""
    # 基础复杂度为1
    complexity = 1
    
    # 决策点计数
    decision_patterns = [
        r'\bif\s*\(',
        r'\belse\s+if\s*\(',
        r'\bfor\s*\(',
        r'\bwhile\s*\(',
        r'\bdo\s*\{',
        r'\bcase\s+\w+:',
        r'\bcatch\s*\(',
        r'\?\s*\w+\s*:',
        r'\|\|',
        r'\&\&',
    ]
    
    for pattern in decision_patterns:
        complexity += len(re.findall(pattern, code, re.MULTILINE))
    
    return complexity


def calculate_cognitive_complexity(code: str) -> int:
    """计算认知复杂度"""
    complexity = 0
    nesting_level = 0
    lines = code.split('\n')
    
    for line in lines:
        stripped = line.strip()
        
        # 减少嵌套层级
        nesting_level -= stripped.count('}')
        
        # 结构增量
        if re.match(r'\b(if|else if|for|while|catch)\b', stripped):
            complexity += 1 + nesting_level
        elif re.match(r'\b(switch|try)\b', stripped):
            complexity += 1
        elif '?' in stripped and ':' in stripped:
            complexity += 1
        
        # 增加嵌套层级
        nesting_level += stripped.count('{')
    
    return complexity


def get_complexity_rating(cyclomatic: int) -> str:
    """获取复杂度评级"""
    if cyclomatic <= 10:
        return "low"
    elif cyclomatic <= 20:
        return "moderate"
    elif cyclomatic <= 50:
        return "high"
    else:
        return "very_high"


def generate_recommendations(cyclomatic: int, cognitive: int, code: str) -> list[str]:
    """生成优化建议"""
    recommendations = []
    
    if cyclomatic > 20:
        recommendations.append(
            f"圈复杂度过高({cyclomatic})，建议提取方法或使用策略模式重构"
        )
    
    if cognitive > 15:
        recommendations.append(
            f"认知复杂度过高({cognitive})，建议减少嵌套层级，简化条件逻辑"
        )
    
    # 检查特定反模式
    if code.count('if') > 5:
        recommendations.append("条件判断过多，考虑使用多态或查找表替代")
    
    if '&&' in code or '||' in code:
        recommendations.append("使用逻辑运算符，确保条件表达式清晰可读")
    
    return recommendations


def main():
    """主入口"""
    try:
        # 从stdin读取输入
        input_data = json.load(sys.stdin)
        
        code = input_data.get("source_code", "")
        file_path = input_data.get("file_path", "")
        method_name = input_data.get("method_name", "")
        
        # 计算方法复杂度
        cyclomatic = calculate_cyclomatic_complexity(code)
        cognitive = calculate_cognitive_complexity(code)
        
        # 计算嵌套深度
        nesting_depth = code.count('{') - code.count('}')
        
        # 获取评级和建议
        rating = get_complexity_rating(cyclomatic)
        recommendations = generate_recommendations(cyclomatic, cognitive, code)
        
        # 输出结果
        result = {
            "cyclomatic_complexity": cyclomatic,
            "cognitive_complexity": cognitive,
            "nesting_depth": nesting_depth,
            "complexity_rating": rating,
            "recommendations": recommendations,
            "file_path": file_path,
            "method_name": method_name
        }
        
        print(json.dumps(result, ensure_ascii=False))
        
    except Exception as e:
        error_result = {
            "error": str(e),
            "error_type": type(e).__name__
        }
        print(json.dumps(error_result, ensure_ascii=False))
        sys.exit(1)


if __name__ == "__main__":
    main()
```

## 七、实施路线图

### 第一阶段：基础优化（1-2周）

1. **启用所有核心专家**
   - correctness_business
   - correctness_technical
   - security_compliance
   - performance_reliability

2. **配置知识规则**
   - 导入Spring/Java核心规则集
   - 启用规则预筛查

3. **调整置信度阈值**
   - 根据团队接受度调整P0-P3阈值
   - 配置低风险提示过滤

### 第二阶段：深度优化（2-4周）

1. **定制专家Prompt**
   - 根据团队编码规范定制审查指南
   - 添加业务特定检查点

2. **扩展上下文分析**
   - 启用完整的DDD上下文组装
   - 配置事务边界分析深度

3. **实施设计一致性检查**
   - 上传详细设计文档模板
   - 配置design-consistency-check Skill

### 第三阶段：持续改进（持续）

1. **收集反馈**
   - 建立人工反馈流程
   - 定期Review误报/漏报案例

2. **优化规则**
   - 根据反馈调整规则权重
   - 添加新的业务特定规则

3. **扩展工具**
   - 开发业务特定的分析工具
   - 集成外部代码质量平台
